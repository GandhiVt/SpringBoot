-- 1. Crear la base de datos (si no existe)
CREATE DATABASE IF NOT EXISTS txair_db; 

-- 2. Usar la base de datos
USE txair_db;

-- 3. Crear la tabla productos
CREATE TABLE IF NOT EXISTS productos (
    id INT PRIMARY KEY AUTO_INCREMENT,
    nombre VARCHAR(100) NOT NULL,
    precio DECIMAL(10, 2) NOT NULL, -- Usamos DECIMAL para dinero, es más preciso
    stock INT NOT NULL
);

-- 4. Insertar datos de ejemplo
-- Borramos datos anteriores para evitar duplicados si ejecutas esto de nuevo
TRUNCATE TABLE productos;
